/\*\*

- @brief 仿真运动学限制参数类
- @details 定义机器人在仿真环境中的速度、加速度上限及控制阈值，
-          用于运动规划时的约束检查和底层控制器的安全保护。
-          所有参数均为编译期常量，运行时不可修改。

  \*/
  class SimLimit
  {
  public:
  // ==================== 速度上限 ====================

      /**
       * @brief 最大线速度 (m/s)
       * @details 机器人底盘平移速度的绝对值上限。
       *          超过此值的速度指令将被截断(clamp)到该值。
       */
      const double max_linear_vel_ = 5.0;

      /**
       * @brief 最大角速度 (rad/s)
       * @details 机器人绕自身Z轴旋转的角速度绝对值上限。
       *          对应约 343.8°/s，超过此值的旋转指令将被截断。
       */
      const double max_angular_vel_ = 6;

      // ==================== 加速度上限 ====================

      /**
       * @brief 最大线加速度 (m/s²)
       * @details 机器人平移加速度的绝对值上限。
       *          用于梯形/S形速度规划中的加速段约束，
       *          防止电机过流或轮胎打滑。
       */
      const double max_acc_linear_ = 2.5;

      /**
       * @brief 最大角加速度 (rad/s²)
       * @details 机器人旋转角加速度的绝对值上限。
       *          限制转向时的惯性冲击，保护陀螺仪和机械结构。
       */
      const double max_acc_angular_ = 3;

      // ==================== 控制阈值 ====================

      /**
       * @brief 加速度触发阈值 (m/s²)
       * @details 当目标加速度 ≥ 此值时，控制器进入"急加速"模式，
       *          可能切换PID参数或启用前馈补偿以提升响应速度。
       *          注意：此值不应超过 max_acc_linear_。
       */
      const double acc_thresh = 2.5;

      /**
       * @brief 减速度触发阈值 (m/s²)
       * @details 当需要减速且减速度绝对值 ≥ 此值时，触发"急刹车"逻辑。
       *          通常 dcc_thresh > acc_thresh，因为制动能力一般强于加速能力。
       *          用于紧急避障或快速停球等场景。
       */
      const double dcc_thresh = 5;

      /**
       * @brief 速度容差阈值 (m/s)
       * @details 当实际速度与目标速度之差 ≤ 此值时，
       *          判定为"已到达目标速度"，可用于状态机切换或轨迹跟踪完成判断。
       *          也作为低速死区，避免微小速度指令导致电机抖动。
       */
      const double speed_thresh = 5;

  };
